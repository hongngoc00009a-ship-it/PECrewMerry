  <script>
    // Small utilities
    const $ = id => document.getElementById(id);
    function updatePreview(){
      const t = $('inputTitle').value.trim() || 'Giáng Sinh ấm áp — 2025';
      const m = $('inputMsg').value.trim() || '';
      const peopleRaw = $('inputPeople').value.trim();
      const people = peopleRaw ? peopleRaw.split(',').map(s=>s.trim()).filter(Boolean) : ['Thầy/Cô','Bạn bè'];
      $('pvTitle').textContent = t;
      $('pvMsg').textContent = m;
      $('pvNames').textContent = 'Gửi: ' + people.join(', ');
      $('greetingTitle').textContent = t;
      $('greetingSub').textContent = m || 'Gửi tới thầy cô và các bạn Khoa Kinh tế — ngành Truyền thông & Tổ chức Sự kiện, chúc Giáng Sinh an lành.';

      // update chips
      const container = $('recipients'); container.innerHTML='';
      people.slice(0,8).forEach(p=>{
        const sp = document.createElement('span'); sp.className='chip'; sp.textContent=p; container.appendChild(sp);
      });
    }

    // Snow generator (creates CSS snowflakes)
    function createSnow(num=40){
      const snow = document.querySelector('.snow'); snow.innerHTML='';
      for(let i=0;i<num;i++){
        const el = document.createElement('div');
        el.className='flake';
        const size = 6 + Math.random()*10;
        el.style.position='absolute';
        el.style.left = Math.random()*100 + '%';
        el.style.top = (-10 - Math.random()*20) + '%';
        el.style.width = size+'px'; el.style.height = size+'px';
        el.style.background = 'radial-gradient(circle at 30% 30%, #fff, rgba(255,255,255,0.85))';
        el.style.borderRadius='50%';
        el.style.opacity = 0.7 + Math.random()*0.3;
        el.style.filter = 'blur('+ (Math.random()*0.6) +'px)';
        el.style.transform = 'translateY(0)';
        el.style.zIndex = 0;
        el.style.animation = 'fall '+(8+Math.random()*12)+'s linear '+(Math.random()*5)+'s infinite';
        snow.appendChild(el);
      }
    }

    // confetti pieces
    function launchConfetti(){
      const container = document.createElement('div'); container.className='confetti'; document.body.appendChild(container);
      const colors = ['#ff4d4d','#ffd166','#74c69d','#4dabf7','#b197fc'];
      for(let i=0;i<80;i++){
        const p = document.createElement('div'); p.className='piece';
        p.style.left = Math.random()*100 + '%';
        p.style.top = Math.random()*40 + '%';
        p.style.background = colors[Math.floor(Math.random()*colors.length)];
        p.style.transform = 'rotate('+ (Math.random()*360) +'deg)';
        p.style.opacity = 1;
        container.appendChild(p);
        // animate
        const dur = 1200 + Math.random()*1400;
        p.animate([
          {transform: 'translateY(0) rotate(0deg)', opacity:1},
          {transform: 'translateY('+ (200 + Math.random()*300) +'px) rotate('+ (Math.random()*720) +'deg)', opacity:0}
        ],{duration:dur, easing:'cubic-bezier(.17,.67,.34,1)', delay: Math.random()*200})
      }
      setTimeout(()=>container.remove(),2500);
    }

    document.querySelectorAll('button').forEach(b=>b.addEventListener('click',e=>e.preventDefault()));
    $('btnUpdate').addEventListener('click', ()=>{ updatePreview(); });
    $('btnConfetti').addEventListener('click', ()=>{ launchConfetti(); });
    $('btnPrint').addEventListener('click', ()=>{ window.print(); });

    // initial
    createSnow(30);
    updatePreview();

    // CSS animation keyframes injected (keeps html single-file)
    const style = document.createElement('style'); style.innerHTML = `@keyframes fall{0%{transform:translateY(-10vh)}100%{transform:translateY(110vh)}}`;
    document.head.appendChild(style);

    // Accessibility: enable keyboard shortcut (Ctrl/Cmd+B) to toggle confetti
    window.addEventListener('keydown', e=>{ if((e.ctrlKey||e.metaKey) && e.key.toLowerCase()==='b'){ e.preventDefault(); launchConfetti(); } });

    // Optional: support quick-add sample names by double-clicking preview
    $('previewCard').addEventListener('dblclick', ()=>{
      const sample = prompt('Thêm danh sách tên (ngăn cách bằng dấu phẩy):','Thầy Hùng, Cô Lan, Nhóm TRUYỂN THÔNG');
      if(sample!=null){ $('inputPeople').value = sample; updatePreview(); }
    });

  </script>
